-- Safely fetch and execute remote mod detector to avoid nil loadstring crashes
local loader = loadstring or load
local function safeRequire(url)
    if not loader then
        warn("Executor has no loadstring; skipping remote script: " .. url)
        return
    end

    local okFetch, srcOrErr = pcall(game.HttpGet, game, url)
    if not okFetch then
        warn("Failed to fetch remote script: " .. tostring(srcOrErr))
        return
    end

    local fn, compileErr = loader(srcOrErr)
    if not fn then
        warn("Failed to compile remote script: " .. tostring(compileErr))
        return
    end

    local okRun, runErr = pcall(fn)
    if not okRun then
        warn("Remote script error: " .. tostring(runErr))
    end
end

-- Function to get the WalkSpeed attribute from humanoid
local function GetBaseWalkspeedFromAttribute(humanoid)
    local attrSpeed = humanoid:GetAttribute("WalkSpeed")
    if attrSpeed then
        return attrSpeed
    end
    return humanoid.WalkSpeed
end

-- Function to consistently apply walkspeed multiplier
local function ApplyWalkspeedMultiplier()
    local character = LocalPlayer.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end
    
    if Config.ShowWalkspeed then
        local baseSpeed = GetBaseWalkspeedFromAttribute(humanoid)
        local targetSpeed = baseSpeed * Config.WalkspeedMultiplier
        
        if math.abs(humanoid.WalkSpeed - targetSpeed) > 0.1 then
            humanoid.WalkSpeed = targetSpeed
        end
    end
end

-- Function to update base walkspeed from attribute
local function UpdateBaseWalkspeed()
    local character = LocalPlayer.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end
    
    BaseWalkspeed = GetBaseWalkspeedFromAttribute(humanoid)
end

-- Start monitoring walkspeed changes
local function StartWalkspeedMonitoring()
    if WalkspeedUpdateConnection then
        WalkspeedUpdateConnection:Disconnect()
        WalkspeedUpdateConnection = nil
    end
    
    local character = LocalPlayer.Character
    if not character then return end
    
    local humanoid = character:FindFirstChild("Humanoid")
    if not humanoid then return end
    
    BaseWalkspeed = GetBaseWalkspeedFromAttribute(humanoid)
    
    local attributeConnection = humanoid:GetAttributeChangedSignal("WalkSpeed"):Connect(function()
        task.wait(0.1)
        BaseWalkspeed = GetBaseWalkspeedFromAttribute(humanoid)
        if Config.ShowWalkspeed then
            ApplyWalkspeedMultiplier()
        end
    end)
    
    WalkspeedUpdateConnection = RunService.RenderStepped:Connect(function()
        if not Config.Enabled or not Config.ShowWalkspeed then return end
        
        local currentChar = LocalPlayer.Character
        if not currentChar then return end
        
        local currentHumanoid = currentChar:FindFirstChild("Humanoid")
        if not currentHumanoid then return end
        
        local currentBase = GetBaseWalkspeedFromAttribute(currentHumanoid)
        
        if math.abs(currentBase - BaseWalkspeed) > 0.1 then
            BaseWalkspeed = currentBase
        end
        
        local currentSpeed = currentHumanoid.WalkSpeed
        local targetSpeed = BaseWalkspeed * Config.WalkspeedMultiplier
        
        if math.abs(currentSpeed - targetSpeed) > 0.1 then
            currentHumanoid.WalkSpeed = targetSpeed
        end
    end)
    
    character.AncestryChanged:Connect(function()
        if not character.Parent then
            if attributeConnection then
                attributeConnection:Disconnect()
            end
        end
    end)
end

-- Handle character respawns
local function OnCharacterAdded(character)
    local humanoid = character:WaitForChild("Humanoid")
    task.wait(0.1)
    
    BaseWalkspeed = GetBaseWalkspeedFromAttribute(humanoid)
    
    if Config.ShowWalkspeed then
        humanoid.WalkSpeed = BaseWalkspeed * Config.WalkspeedMultiplier
    end
    
    StartWalkspeedMonitoring()
    
    humanoid.AncestryChanged:Connect(function()
        task.wait(0.1)
        if not humanoid.Parent then
            task.wait(0.5)
            OnCharacterAdded(character)
        end
    end)
end

-- Function to save configs to file
local function SaveConfigs()
    local success, err = pcall(function()
        local data = {
            CurrentConfig = CurrentConfig,
            Configs = Configs
        }
        writefile(ConfigFolder .. "/" .. ConfigFile, game:GetService("HttpService"):JSONEncode(data))
    end)
    if not success then
        warn("Failed to save configs: " .. tostring(err))
    end
end

-- Function to load configs from file
local function LoadConfigs()
    local success, err = pcall(function()
        if isfile(ConfigFolder .. "/" .. ConfigFile) then
            local data = game:GetService("HttpService"):JSONDecode(readfile(ConfigFolder .. "/" .. ConfigFile))
            CurrentConfig = data.CurrentConfig or "Default"
            Configs = data.Configs or {}
            
            if Configs[CurrentConfig] then
                for k, v in pairs(Configs[CurrentConfig]) do
                    Config[k] = v
                end
            end
        else
            Configs["Default"] = {
                Enabled = true,
                ShowNames = true,
                ShowBars = true,
                ShowNPCs = true,
                ShowValues = true,
                ShowWalkspeed = true,
                WalkspeedMultiplier = 1.0,
                MaxDistance = 1500,
                TextSize = 14,
                BarWidth = 150,
                BarHeight = 6,
                YOffset = 60
            }
            CurrentConfig = "Default"
            SaveConfigs()
        end
    end)
    if not success then
        warn("Failed to load configs: " .. tostring(err))
        Configs["Default"] = {
            Enabled = true,
            ShowNames = true,
            ShowBars = true,
            ShowNPCs = true,
            ShowValues = true,
            ShowWalkspeed = true,
            WalkspeedMultiplier = 1.0,
            MaxDistance = 1500,
            TextSize = 14,
            BarWidth = 150,
            BarHeight = 6,
            YOffset = 60
        }
        CurrentConfig = "Default"
    end
end

local function CreateDrawing(type, props)
    local obj = Drawing.new(type)
    for k, v in pairs(props) do obj[k] = v end
    return obj
end

local function RemoveESP(model)
    if ESP_Cache[model] then
        for _, drawing in pairs(ESP_Cache[model].Drawings) do
            drawing:Remove()
        end
        ESP_Cache[model] = nil
    end
end

RunService.RenderStepped:Connect(function()
    if not Config.Enabled then 
        for model, _ in pairs(ESP_Cache) do RemoveESP(model) end
        return 
    end

    if not LiveFolder then 
        LiveFolder = workspace:FindFirstChild("Live") 
        return 
    end

    for _, entry in pairs(ESP_Cache) do 
        entry.Seen = false 
    end

    for _, folder in ipairs(LiveFolder:GetChildren()) do
        local model = folder:IsA("Model") and folder or folder:FindFirstChildWhichIsA("Model")
        
        if model and model ~= LocalPlayer.Character then
            
            local head = model:FindFirstChild("Head")
            local humanoid = model:FindFirstChild("Humanoid")
            local player = Players:GetPlayerFromCharacter(model)

            -- === PLAYER LOGIC (AlwaysOn Health Bar) ===
            if player and humanoid then
                pcall(function()
                    humanoid.HealthDisplayType = Enum.HumanoidHealthDisplayType.AlwaysOn
                end)
            end

            local isNPC = not player

            -- Check if we should skip due to settings
            if isNPC and not Config.ShowNPCs then
                if ESP_Cache[model] then RemoveESP(model) end
                continue
            end

            if head and humanoid then
                if not ESP_Cache[model] then
                    ESP_Cache[model] = {
                        Seen = true,
                        Drawings = {
                            -- Name
                            Name = CreateDrawing("Text", {Center = true, Outline = true, Color = Color3.new(1,1,1), Size = Config.TextSize}),
                            
                            -- Bar1: Health/Lifeforce (Players: Red, NPCs: Green)
                            Bar1Bg = CreateDrawing("Square", {Filled = true, Color = Color3.new(0.1,0.1,0.1), Transparency = 0.8}),
                            Bar1Fill = CreateDrawing("Square", {Filled = true, Transparency = 1}),
                            Bar1Text = CreateDrawing("Text", {Center = false, Outline = true, Color = Color3.new(1,1,1), Size = 13}),
                            
                            -- Bar2: Posture (Players only - Yellow)
                            Bar2Bg = CreateDrawing("Square", {Filled = true, Color = Color3.new(0.1,0.1,0.1), Transparency = 0.8}),
                            Bar2Fill = CreateDrawing("Square", {Filled = true, Transparency = 1}),
                            Bar2Text = CreateDrawing("Text", {Center = false, Outline = true, Color = Color3.new(1,1,1), Size = 13}),
                        }
                    }
                end

                local entry = ESP_Cache[model]
                entry.Seen = true
                local d = entry.Drawings

                local vector, onScreen = Camera:WorldToViewportPoint(head.Position)
                local distance = (Camera.CFrame.Position - head.Position).Magnitude

                if onScreen and distance <= Config.MaxDistance then
                    
                    -- Initialize values
                    local val1, min1, max1, color1 = 0, 0, 100, Color3.new(1, 0, 0) -- Default red
                    local val2, min2, max2, color2 = 0, 0, 100, Color3.new(1, 1, 0) -- Default yellow
                    local showBar2 = false
                    local cleanName = ""

                    if isNPC then
                        -- NPC LOGIC: Green bar only, NO NAME
                        val1 = humanoid.Health
                        max1 = humanoid.MaxHealth
                        min1 = 0
                        color1 = Color3.new(0.2, 0.9, 0.2) -- Green
                        showBar2 = false
                        cleanName = "" -- NO NAME FOR NPCs
                    else
                        -- PLAYER LOGIC: Show name + both bars (Lifeforce and Posture)
                        cleanName = (player.DisplayName and player.DisplayName ~= "" and player.DisplayName) or player.Name

                        -- Look for Lifeforce (Bar1)
                        local lf = folder:FindFirstChild("Lifeforce") or model:FindFirstChild("Lifeforce")
                        if lf then 
                            val1 = lf.Value 
                            max1 = lf:GetAttribute("MaxValue") or lf.MaxValue or 100
                            min1 = lf:GetAttribute("MinValue") or lf.MinValue or 0
                        else
                            -- Fallback to humanoid health
                            val1 = humanoid.Health
                            max1 = humanoid.MaxHealth
                            min1 = 0
                        end
                        
                        -- Look for Posture (Bar2)
                        local pos = folder:FindFirstChild("Posture") or model:FindFirstChild("Posture")
                        if pos then 
                            val2 = pos.Value 
                            max2 = pos:GetAttribute("MaxValue") or pos.MaxValue or 100
                            min2 = pos:GetAttribute("MinValue") or pos.MinValue or 0
                            showBar2 = true
                        end
                    end
                    
                    -- Calculate Percentages
                    local range1 = max1 - min1
                    local pct1 = range1 > 0 and math.clamp((val1 - min1) / range1, 0, 1) or 0
                    
                    local range2 = max2 - min2
                    local pct2 = range2 > 0 and math.clamp((val2 - min2) / range2, 0, 1) or 0

                    local screenPos = Vector2.new(vector.X, vector.Y)
                    
                    -- Calculate starting Y position (bars will stack upward from here)
                    local startY = screenPos.Y - Config.YOffset
                    
                    -- Current Y position for stacking (start from bottom and go up)
                    local currentY = startY

                    -- NAME (Players only)
                    if Config.ShowNames and not isNPC and cleanName ~= "" then
                        d.Name.Visible = true
                        d.Name.Text = cleanName
                        -- Name above all bars
                        d.Name.Position = Vector2.new(screenPos.X, startY - 30)
                        d.Name.Size = Config.TextSize
                    else
                        d.Name.Visible = false
                    end

                    -- BARS
                    if Config.ShowBars then
                        -- Determine bar order (bottom to top: Health/Lifeforce first, then Posture above it)
                        
                        -- Bar1: Health/Lifeforce (bottom bar)
                        local bar1Pos = Vector2.new(screenPos.X - (Config.BarWidth / 2), currentY)
                        
                        d.Bar1Bg.Visible = true
                        d.Bar1Bg.Size = Vector2.new(Config.BarWidth, Config.BarHeight)
                        d.Bar1Bg.Position = bar1Pos
                        d.Bar1Fill.Visible = true
                        d.Bar1Fill.Size = Vector2.new(Config.BarWidth * pct1, Config.BarHeight)
                        d.Bar1Fill.Position = bar1Pos
                        d.Bar1Fill.Color = color1

                        if Config.ShowValues then
                            d.Bar1Text.Visible = true
                            d.Bar1Text.Text = string.format("%.0f/%.0f", val1, max1)
                            d.Bar1Text.Position = bar1Pos + Vector2.new(Config.BarWidth + 5, -3)
                        else
                            d.Bar1Text.Visible = false
                        end
                        
                        -- Move up for next bar
                        currentY = currentY - (Config.BarHeight + 2)
                        
                        -- Bar2: Posture (above health bar) - only for players
                        if showBar2 and not isNPC then
                            local bar2Pos = Vector2.new(screenPos.X - (Config.BarWidth / 2), currentY)
                            
                            d.Bar2Bg.Visible = true
                            d.Bar2Bg.Size = Vector2.new(Config.BarWidth, Config.BarHeight)
                            d.Bar2Bg.Position = bar2Pos
                            d.Bar2Fill.Visible = true
                            d.Bar2Fill.Size = Vector2.new(Config.BarWidth * pct2, Config.BarHeight)
                            d.Bar2Fill.Position = bar2Pos
                            d.Bar2Fill.Color = color2

                            if Config.ShowValues then
                                d.Bar2Text.Visible = true
                                d.Bar2Text.Text = string.format("%.0f/%.0f", val2, max2)
                                d.Bar2Text.Position = bar2Pos + Vector2.new(Config.BarWidth + 5, -3)
                            else
                                d.Bar2Text.Visible = false
                            end
                        else
                            d.Bar2Bg.Visible = false
                            d.Bar2Fill.Visible = false
                            d.Bar2Text.Visible = false
                        end
                    else
                        -- Hide all bars
                        d.Bar1Bg.Visible = false
                        d.Bar1Fill.Visible = false
                        d.Bar1Text.Visible = false
                        d.Bar2Bg.Visible = false
                        d.Bar2Fill.Visible = false
                        d.Bar2Text.Visible = false
                    end

                else
                    -- Hide everything if off screen or too far
                    for _, v in pairs(d) do 
                        if v and v.Visible ~= nil then
                            v.Visible = false 
                        end
                    end
                end
            end
        end
    end

    -- Clean up ESP for models that are no longer valid
    for model, entry in pairs(ESP_Cache) do
        if not entry.Seen or not model or not model.Parent then
            RemoveESP(model)
        end
    end
end)

    local markers = {

        print("Found main missionMarker at: " .. tostring(mainMarker.Position))
    end
    
    -- Look for the second missionMarker (in train station)
    local trainStation = workspace:FindFirstChild("townZones") and 
                         workspace.townZones:FindFirstChild("Minami Village Train Station")
    if trainStation then
        local customMarkers = trainStation:FindFirstChild("CustomMarkers")
        if customMarkers then
            local stationMarker = customMarkers:FindFirstChild("missionMarker")
            if stationMarker and stationMarker:IsA("BasePart") then
                markers.missionMarker2 = stationMarker
                print("Found station missionMarker at: " .. tostring(stationMarker.Position))
            end
        end
    end
    
    -- Look for white marker (start line) - adjust as needed
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") then
            -- Check for white marker
            if obj.BrickColor == BrickColor.new("White") or 
               obj.Color == Color3.new(1, 1, 1) or
               obj.Name:lower():find("white") or
               obj.Name:lower():find("start") then
                markers.white = obj
                print("Found white marker at: " .. tostring(obj:GetFullName()))
                break
            end
        end
    end
    
    return markers
end


-- Function to find nearest NPC to player
local function FindNearestNPC(player)
    local character = player.Character
    if not character then return nil, math.huge end
    
    local rootPart = character:FindFirstChild("HumanoidRootPart")
    if not rootPart then return nil, math.huge end
    
    local playerPos = rootPart.Position
    local nearestNPC = nil
    local shortestDist = math.huge
    
    if not LiveFolder then return nil, math.huge end
    
    for _, npc in ipairs(LiveFolder:GetChildren()) do
        -- Skip if it's a player character
        if Players:GetPlayerFromCharacter(npc) then continue end
        
        local npcRoot = npc:FindFirstChild("HumanoidRootPart") or npc.PrimaryPart
        if npcRoot then
            local dist = (npcRoot.Position - playerPos).Magnitude
            if dist < shortestDist then
                shortestDist = dist
                nearestNPC = npc
            end
        end
    end
    
    return nearestNPC, shortestDist
end

-- Function to make NPC follow the player
local function MakeNPCFollow(npc, player)
    if not npc or not player then return end
    
    -- Stop any existing follow connection
    if EscortMission.FollowConnection then
        EscortMission.FollowConnection:Disconnect()
        EscortMission.FollowConnection = nil
    end
    
    local humanoid = npc:FindFirstChild("Humanoid")
    local rootPart = npc:FindFirstChild("HumanoidRootPart") or npc.PrimaryPart
    
    if not humanoid or not rootPart then 
        Library:Notify("NPC can't follow (missing Humanoid)", 3)
        return 
    end
    
    EscortMission.Following = true
    
    -- Make NPC follow player
    EscortMission.FollowConnection = RunService.Heartbeat:Connect(function()
        if not EscortMission.Active or not EscortMission.Following then
            return
        end
        
        -- Check if NPC and player still exist
        if not npc or not npc.Parent or not player.Character then
            EscortMission.Following = false
            return
        end
        
        local playerChar = player.Character
        local playerRoot = playerChar:FindFirstChild("HumanoidRootPart")
        
        if not playerRoot then return end
        
        -- Calculate distance between NPC and player
        local npcPos = rootPart.Position
        local playerPos = playerRoot.Position
        local distance = (npcPos - playerPos).Magnitude
        
        -- If NPC is too far, move closer
        if distance > 8 then
            humanoid:MoveTo(playerPos)
        elseif distance < 4 then
            -- If too close, back up a little
            local direction = (npcPos - playerPos).Unit
            local backupPos = playerPos + direction * 5
            humanoid:MoveTo(backupPos)
        else
            -- Just stop moving if at good distance
            humanoid:MoveTo(npcPos) -- Stop moving
        end
        
        -- Check if we've reached the red marker (mission complete)
        local markers = FindMarkers()
        if markers.red then
            local distToRed = (rootPart.Position - markers.red.Position).Magnitude
            if distToRed < 15 then -- Within 15 studs of red marker
                -- Auto-complete the mission
                CompleteEscortMission(true) -- Pass true for "auto complete"
            end
        end
    end)
end

-- Function to start escort mission
local function StartEscortMission()
    if EscortMission.Active then
        Library:Notify("Mission already active!", 3)
        return
    end
    
    local player = LocalPlayer
    local nearestNPC, distance = FindNearestNPC(player)
    
    if not nearestNPC then
        Library:Notify("No NPCs found nearby!", 3)
        return
    end
    
    if distance > 30 then -- Only allow if within 30 studs
        Library:Notify("NPC too far! Get closer (within 30 studs)", 3)
        return
    end
    
    local markers = FindMarkers()
    if not markers.red then
        Library:Notify("Red marker (finish line) not found in workspace!", 3)
        return
    end
    
    if not markers.white then
        Library:Notify("White marker (start line) not found!", 2)
        -- Continue anyway, just warn
    end
    
    -- Check if we're at the start line (white marker)
    if markers.white then
        local playerChar = player.Character
        if playerChar and playerChar:FindFirstChild("HumanoidRootPart") then
            local distToStart = (playerChar.HumanoidRootPart.Position - markers.white.Position).Magnitude
            if distToStart > 20 then
                Library:Notify("Go to the white marker to start the mission!", 3)
                return
            end
        end
    end
    
    -- Start the mission
    EscortMission.Active = true
    EscortMission.CurrentNPC = nearestNPC
    EscortMission.CurrentPlayer = player
    EscortMission.RedMarker = markers.red
    EscortMission.WhiteMarker = markers.white
    EscortMission.MissionStartTime = tick()
    
    -- Get NPC info for display
    local npcName = nearestNPC.Name
    
    Library:Notify(string.format("Escort mission started! Lead %s to the red marker!", npcName), 3)
    
    -- Highlight the NPC
    if nearestNPC:FindFirstChild("Highlight") then
        nearestNPC.Highlight.Enabled = true
        nearestNPC.Highlight.FillColor = Color3.new(0, 1, 0)
        nearestNPC.Highlight.OutlineColor = Color3.new(1, 1, 1)
    else
        local highlight = Instance.new("Highlight")
        highlight.Parent = nearestNPC
        highlight.FillColor = Color3.new(0, 1, 0)
        highlight.OutlineColor = Color3.new(1, 1, 1)
        highlight.Enabled = true
    end
    
    -- Make the NPC follow
    MakeNPCFollow(nearestNPC, player)
end

-- Function to complete escort mission (with optional auto-complete)
local function CompleteEscortMission(autoComplete)
    if not EscortMission.Active then
        Library:Notify("No active mission!", 3)
        return
    end
    
    if not EscortMission.CurrentNPC or not EscortMission.CurrentNPC.Parent then
        Library:Notify("NPC is no longer available!", 3)
        EscortMission.Active = false
        return
    end
    
    local markers = FindMarkers()
    if not markers.red then
        Library:Notify("Red marker not found!", 3)
        return
    end
    
    -- Stop following
    if EscortMission.FollowConnection then
        EscortMission.FollowConnection:Disconnect()
        EscortMission.FollowConnection = nil
    end
    EscortMission.Following = false
    
    local timeTaken = math.floor(tick() - EscortMission.MissionStartTime)
    
    if autoComplete then
        Library:Notify(string.format("Mission complete! You reached the finish line in %d seconds!", timeTaken), 5)
    else
        Library:Notify(string.format("Mission completed manually! Time: %d seconds", timeTaken), 5)
    end
    
    -- Remove highlight
    if EscortMission.CurrentNPC:FindFirstChild("Highlight") then
        EscortMission.CurrentNPC.Highlight.Enabled = false
    end
    
    -- Reset mission state
    EscortMission.Active = false
    EscortMission.CurrentNPC = nil
    EscortMission.Following = false
end

-- Function 2: Teleport both player and NPC 300 studs towards white marker
local function TeleportTowardsWhiteMarker()
    if not EscortMission.Active then
        Library:Notify("No active mission to teleport!", 3)
        return
    end
    
    local markers = FindMarkers()
    
    if not markers.white then
        Library:Notify("White marker not found! Can't calculate direction.", 3)
        return
    end
    
    -- Get current positions
    local player = LocalPlayer
    local playerRoot = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    local npcRoot = EscortMission.CurrentNPC and (EscortMission.CurrentNPC:FindFirstChild("HumanoidRootPart") or EscortMission.CurrentNPC.PrimaryPart)
    
    if not playerRoot or not npcRoot then
        Library:Notify("Player or NPC not found!", 3)
        return
    end
    
    -- Calculate direction from current position to white marker
    local currentPos = npcRoot.Position
    local whitePos = markers.white.Position
    
    -- Vector from current position to white marker
    local directionToWhite = (whitePos - currentPos).Unit
    
    -- Calculate new position 300 studs towards white marker
    local distanceToWhite = (whitePos - currentPos).Magnitude
    local teleportDistance = math.min(300, distanceToWhite - 10) -- Don't overshoot, leave 10 studs buffer
    
    if teleportDistance < 10 then
        Library:Notify("Already too close to white marker!", 3)
        return
    end
    
    local newPosition = currentPos + (directionToWhite * teleportDistance)
    
    print(string.format("Teleporting 300 studs towards white marker"))
    print(string.format("From: %.0f, %.0f, %.0f", currentPos.X, currentPos.Y, currentPos.Z))
    print(string.format("To: %.0f, %.0f, %.0f", newPosition.X, newPosition.Y, newPosition.Z))
    print(string.format("Distance to white: %.0f studs", (whitePos - newPosition).Magnitude))
    
    -- Teleport NPC
    npcRoot.CFrame = CFrame.new(newPosition) * CFrame.new(0, 0, 2)
    
    -- Teleport Player (position them slightly behind the NPC)
    playerRoot.CFrame = CFrame.new(newPosition) * CFrame.new(0, 0, -4)
    
    -- Add teleport effect
    local effect = Instance.new("Part")
    effect.Size = Vector3.new(8, 8, 8)
    effect.CFrame = CFrame.new(newPosition)
    effect.Anchored = true
    effect.CanCollide = false
    effect.Transparency = 0.5
    effect.BrickColor = BrickColor.new("Bright green")
    effect.Parent = workspace
    
    -- Fade out effect
    for i = 1, 10 do
        effect.Transparency = effect.Transparency + 0.1
        task.wait(0.03)
    end
    
    effect:Destroy()
    
    Library:Notify(string.format("Teleported 300 studs towards white marker! (%.0f studs remaining)", (whitePos - newPosition).Magnitude), 3)
end

-- Function to teleport to in-game location
local function TeleportInGame(locationName, position)
    local player = LocalPlayer
    if not player.Character then
        Library:Notify("No character found!", 3)
        return
    end
    
    local humanoidRootPart = player.Character:FindFirstChild("HumanoidRootPart")
    if not humanoidRootPart then
        Library:Notify("Cannot teleport (no HumanoidRootPart)", 3)
        return
    end
    
    -- Teleport effect
    local effect = Instance.new("Part")
    effect.Size = Vector3.new(5, 5, 5)
    effect.CFrame = humanoidRootPart.CFrame
    effect.Anchored = true
    effect.CanCollide = false
    effect.Transparency = 0.5
    effect.BrickColor = BrickColor.new("Bright blue")
    effect.Parent = workspace
    
    -- Fade out
    for i = 1, 10 do
        effect.Transparency = effect.Transparency + 0.1
        task.wait(0.02)
    end
    
    -- Teleport
    humanoidRootPart.CFrame = CFrame.new(position)
    
    -- Fade in at new location
    effect.CFrame = humanoidRootPart.CFrame
    effect.Transparency = 0.5
    for i = 1, 10 do
        effect.Transparency = effect.Transparency - 0.1
        task.wait(0.02)
    end
    
    effect:Destroy()
    
    Library:Notify(string.format("Teleported to %s!", locationName), 3)
end
